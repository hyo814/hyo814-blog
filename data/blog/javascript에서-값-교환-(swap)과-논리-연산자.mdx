효율적인 연산과 조건 검사를 통해 데이터 처리와 유효성 검증을 간결하고 명확하게 구현.'

```jsx
// JavaScript에서 값 교환 (Swap)과 논리 연산자

/**
 * 1. 값을 교환하는 방법 (Swap)
 */

// 방법 1: 임시 변수 사용
function swapUsingTemp(a, b) {
  let temp = a
  a = b
  b = temp
  return [a, b]
}
console.log('임시 변수를 사용한 값 교환:', swapUsingTemp(3, 5)) // [5, 3]

// 방법 2: 배열 구조 분해 (Destructuring)
function swapUsingDestructuring(a, b) {
  ;[a, b] = [b, a]
  return [a, b]
}
console.log('구조 분해를 사용한 값 교환:', swapUsingDestructuring(3, 5)) // [5, 3]

// 방법 3: XOR 비트 연산자 사용
function swapUsingXOR(a, b) {
  a = a ^ b
  b = a ^ b
  a = a ^ b
  return [a, b]
}
console.log('XOR 비트 연산자를 사용한 값 교환:', swapUsingXOR(3, 5)) // [5, 3]

// 방법 4: 덧셈과 뺄셈을 사용한 값 교환
function swapUsingArithmetic(a, b) {
  a = a + b
  b = a - b
  a = a - b
  return [a, b]
}
console.log('덧셈과 뺄셈을 사용한 값 교환:', swapUsingArithmetic(3, 5)) // [5, 3]

/**
 * 2. 논리 연산자 (Logical Operators)
 */

// 논리 연산자
const a = true
const b = false

// AND 연산자 (&&)
console.log('AND 연산자:', a && b) // false

// OR 연산자 (||)
console.log('OR 연산자:', a || b) // true

// NOT 연산자 (!)
console.log('NOT 연산자:', !a) // false

// 단축 평가 (Short-Circuit Evaluation)
console.log('AND 단축 평가:', a && 'Hello') // "Hello" (a가 true이므로 오른쪽 값 반환)
console.log('OR 단축 평가:', b || 'World') // "World" (b가 false이므로 오른쪽 값 반환)

// Nullish 병합 연산자 (??)
const x = null
const y = 'Default'
console.log('Nullish 병합 연산자:', x ?? y) // "Default" (x가 null이므로 y 반환)

// 삼항 연산자 (Ternary Operator)
const isAdult = true
console.log('삼항 연산자:', isAdult ? '성인' : '미성년자') // "성인"

/**
 * 3. 비트 논리 연산자 (Bitwise Operators)
 */
const num1 = 5 // 0101
const num2 = 3 // 0011

// AND 비트 연산자 (&)
console.log('AND 비트 연산자:', num1 & num2) // 1 (0001)

// OR 비트 연산자 (|)
console.log('OR 비트 연산자:', num1 | num2) // 7 (0111)

// XOR 비트 연산자 (^)
console.log('XOR 비트 연산자:', num1 ^ num2) // 6 (0110)

// NOT 비트 연산자 (~)
console.log('NOT 비트 연산자:', ~num1) // -6 (1111...1010)

// Shift 연산자 (<<, >>)
console.log('Left Shift (<<):', num1 << 1) // 10 (1010)
console.log('Right Shift (>>):', num1 >> 1) // 2 (0010)

/**
 * 4. 논리 연산자와 조건문 활용 예제
 */

// 기본값 설정 (OR 사용)
function greet(name) {
  const userName = name || '익명'
  console.log(`안녕하세요, ${userName}!`)
}
greet() // "안녕하세요, 익명!"
greet('홍길동') // "안녕하세요, 홍길동!"

// 조건부 실행 (AND 사용)
const isLoggedIn = true
isLoggedIn && console.log('사용자가 로그인했습니다.') // 조건이 true이므로 실행

// Nullish 병합 연산자 활용
const userInput = null
const defaultValue = '기본 값'
console.log(userInput ?? defaultValue) // "기본 값"

/**
 * 5. 유효성 검사 활용
 */
const age = 20
if (age >= 18 && age <= 25) {
  console.log('대학생입니다.') // 조건이 true일 경우 실행
}
```
